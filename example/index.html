js.colorGradient

getHexColorAtPercent([0...100])

use result directly as hex color

    var gradient1 = new ColorGradient(["#fff", "#ff00ff", "000"]);
    var hexString = gradient1.getHexColorAtPercent([0...100]);

getColorAtPercent([0...100])

use result as r/b/g

    var gradient2 = new ColorGradient(["f00", "0f0", "00f", "ff0", "f0f", "0ff", "fff", "000"]);
    var rgbArray = gradient2.getColorAtPercent([0...100]);
